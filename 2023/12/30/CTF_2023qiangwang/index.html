    r.sendline(b"2")

    count = 0
    for i in range(80):
        r.recvuntil(b">")
        r.sendline(b"0")
        temp = r.recvline()
        if(b"Right" in temp):

    print(count)
    if(count > 56):
        print(r.recvline())

    ret = 0
